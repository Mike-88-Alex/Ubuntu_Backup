git add - puts in staging area
git add . - puts ALL files in staging area
git add -A - puts ALL unstaged files to the staging area within the repository
git commit -m "Add <ceva aici>" - commit

git remote -v - verifica URL origin

git status - shows status of files
git log - shows files commited and timestamps + author

git push origin main - upload files to github
